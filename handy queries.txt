SELECT t1.id,t4.title as property, t1.from_date,t1.to_date,CONVERT_TZ(t1.created_on,'+00:00','+05:30') as created_on, CONVERT_TZ(t1.updated_on,'+00:00','+05:30') as deleted_on, t2.name_of_user as created_by,t3.name_of_user AS deleted_by FROM `reservation` t1 LEFT JOIN admin_user t2 ON t1.created_by=t2.id LEFT JOIN admin_user t3 ON t1.updated_by=t3.id LEFT JOIN property t4 ON t1.property_id=t4.id where t1.property_id=39 and t1.from_date='2021-09-24'
